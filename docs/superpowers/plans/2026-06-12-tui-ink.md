**Goal:** 用 Ink (React for CLI) 完全重写 CodecastCLI 的 UI 层，实现三段式全屏 TUI。

  green: '\x1b[32m',

  reset: '\x1b[0m',

export function highlightCode(code: string, lang: string): string {
  if (['js', 'ts', 'javascript', 'typescript'].includes(lang)) {
    return code
      .replace(/\b(const|let|var|function|return|if|else|for|while|import|export|from|async|await|class|interface|type|enum|extends|implements|new|this|throw|try|catch|finally)\b/g, `${KEYWORD_COLORS.cyan}$1${KEYWORD_COLORS.reset}`)
      .replace(/('.*?'|".*?"|`.*?`)/g, `${KEYWORD_COLORS.green}$1${KEYWORD_COLORS.reset}`)
      .replace(/(\/\/.*$)/gm, `${KEYWORD_COLORS.gray}$1${KEYWORD_COLORS.reset}`)
  }
  if (['py', 'python'].includes(lang)) {
    return code
      .replace(/\b(def|class|return|if|else|elif|for|while|import|from|as|try|except|with|yield|lambda|pass|raise|finally|async|await)\b/g, `${KEYWORD_COLORS.cyan}$1${KEYWORD_COLORS.reset}`)
      .replace(/('.*?'|".*?"|f'.*?'|f".*?")/g, `${KEYWORD_COLORS.green}$1${KEYWORD_COLORS.reset}`)
      .replace(/(#.*$)/gm, `${KEYWORD_COLORS.gray}$1${KEYWORD_COLORS.reset}`)
  }
  if (['go'].includes(lang)) {
    return code
      .replace(/\b(func|return|if|else|for|range|import|package|var|const|type|struct|interface|switch|case|default|defer|go|chan|select)\b/g, `${KEYWORD_COLORS.cyan}$1${KEYWORD_COLORS.reset}`)
      .replace(/(".*?")/g, `${KEYWORD_COLORS.green}$1${KEYWORD_COLORS.reset}`)
      .replace(/(\/\/.*$)/gm, `${KEYWORD_COLORS.gray}$1${KEYWORD_COLORS.reset}`)
  }
  if (['rs', 'rust'].includes(lang)) {
    return code
      .replace(/\b(fn|let|mut|pub|return|if|else|for|while|loop|match|impl|struct|enum|trait|use|mod|async|await|self|super|where|type|const|static|ref)\b/g, `${KEYWORD_COLORS.cyan}$1${KEYWORD_COLORS.reset}`)

// src/tui/diff.ts
import { DiffLine } from './types.js'

/** 计算两段文本的 diff（从 ToolRenderer 提取） */
export function computeDiff(oldContent: string, newContent: string): DiffLine[] {
  const oldLines = oldContent.split('\n')
  const newLines = newContent.split('\n')
  const result: DiffLine[] = []
  const maxLines = Math.max(oldLines.length, newLines.length)

  for (let i = 0; i < maxLines; i++) {
    const oldLine = oldLines[i]
    const newLine = newLines[i]

    if (oldLine === undefined && newLine !== undefined) {
      result.push({ type: 'add', content: newLine, lineNumber: i + 1 })
    } else if (oldLine !== undefined && newLine === undefined) {
      result.push({ type: 'remove', content: oldLine, lineNumber: i + 1 })
    } else if (oldLine !== newLine) {
      result.push({ type: 'remove', content: oldLine, lineNumber: i + 1 })
      result.push({ type: 'add', content: newLine, lineNumber: i + 1 })
    } else {
      result.push({ type: 'context', content: oldLine, lineNumber: i + 1 })
    }
  }

  return result
}

/** 仅返回有变化的行（上下文各 3 行） */
export function computeDiffCompact(oldContent: string, newContent: string, contextLines = 3): DiffLine[] {
  const full = computeDiff(oldContent, newContent)
  const changedIndices = full
    .map((line, i) => (line.type !== 'context' ? i : -1))
    .filter(i => i >= 0)

  if (changedIndices.length === 0) return []

  const visible = new Set<number>()
  for (const idx of changedIndices) {
    for (let j = Math.max(0, idx - contextLines); j <= Math.min(full.length - 1, idx + contextLines); j++) {
      visible.add(j)
    }
  }

  return full.filter((_, i) => visible.has(i))
}
```

interface CodeBlockProps {
  code: string
  language?: string
  maxWidth?: number
}

export function CodeBlock({ code, language = '', maxWidth = 80 }: CodeBlockProps) {
  const highlighted = language ? highlightCode(code, language) : code
  const lines = highlighted.split('\n')
  const width = Math.min(maxWidth, process.stdout.columns || 80)
  const header = `┌─ ${language || 'code'} ${'─'.repeat(Math.max(0, width - (language || 'code').length - 6))}┐`
  const footer = '└' + '─'.repeat(width - 2) + '┘'

    <Box flexDirection="column">
      <Text color="gray">{header}</Text>
      {lines.map((line, i) => (
        <Box key={i}>
          <Text color="gray">│</Text>
          <Text>{line.slice(0, width - 2)}</Text>
          <Text color="gray">│</Text>
        </Box>
      ))}
      <Text color="gray">{footer}</Text>

  )
}

interface DiffViewProps {

import React from 'react'
import { Box, Text } from 'ink'
import { Message } from './types.js'
import { ToolCallView } from './tool-call-view.js'
import { CodeBlock } from './code-block.js'

interface MessageItemProps {
  message: Message
}

/** 解析 Markdown 中的代码块，返回分段内容 */
function parseContent(content: string): Array<{ type: 'text'; value: string } | { type: 'code'; value: string; lang: string }> {
  const parts: Array<{ type: 'text'; value: string } | { type: 'code'; value: string; lang: string }> = []
  const regex = /```(\w*)\n([\s\S]*?)```/g
  let lastIndex = 0
  let match

  while ((match = regex.exec(content)) !== null) {
    if (match.index > lastIndex) {
      parts.push({ type: 'text', value: content.slice(lastIndex, match.index) })
    }
    parts.push({ type: 'code', value: match[2], lang: match[1] })
    lastIndex = match.index + match[0].length
  }

  if (lastIndex < content.length) {
    parts.push({ type: 'text', value: content.slice(lastIndex) })
  }

  return parts.length > 0 ? parts : [{ type: 'text', value: content }]
}

export function MessageItem({ message }: MessageItemProps) {
  if (message.role === 'user') {
    return (
      <Box>
        <Text color="cyan">You: </Text>
        <Text>{message.content}</Text>
      </Box>
    )
  }

  if (message.role === 'tool_call' || message.role === 'tool_result') {
    return <ToolCallView message={message} />
  }

  // assistant
  const parts = parseContent(message.content)
  return (
    <Box flexDirection="column">
      <Text color="yellow">AI:</Text>
      {parts.map((part, i) => {
        if (part.type === 'code') {
          return <CodeBlock key={i} code={part.value} language={part.lang} />
        }
        return <Text key={i}>{part.value}</Text>
      })}
    </Box>
  )
}

// src/tui/stream-message.tsx
import React from 'react'
import { Box, Text } from 'ink'
import { ActiveMessage } from './types.js'
import { ToolCallView } from './tool-call-view.js'
import { CodeBlock } from './code-block.js'
import Spinner from 'ink-spinner'

interface StreamMessageProps {
  message: ActiveMessage
}

/** 简单的代码块检测：如果内容包含未闭合的 ``` 则正在写代码块 */
function parseStreamContent(content: string): {
  text: string
  inCodeBlock: boolean
  codeContent: string
  codeLang: string
} {
  const codeBlockStarts = (content.match(/```/g) || []).length
  const inCodeBlock = codeBlockStarts % 2 === 1

  if (!inCodeBlock) {
    return { text: content, inCodeBlock: false, codeContent: '', codeLang: '' }
  }

  // 找到最后一个 ``` 的位置
  const lastFence = content.lastIndexOf('```')
  const beforeFence = content.slice(0, lastFence)
  const afterFence = content.slice(lastFence + 3)
  const lang = afterFence.split('\n')[0]?.trim() || ''
  const code = afterFence.slice(lang.length).replace(/^\n/, '')

  return { text: beforeFence, inCodeBlock: true, codeContent: code, codeLang: lang }
}

export function StreamMessage({ message }: StreamMessageProps) {
  const { text, inCodeBlock, codeContent, codeLang } = parseStreamContent(message.content)

  return (
    <Box flexDirection="column">
      <Box>
        <Text color="yellow">AI: </Text>
        <Text color="green"><Spinner type="dots" /></Text>
      </Box>
      {text && <Text>{text}</Text>}
      {inCodeBlock && codeContent && (
        <CodeBlock code={codeContent} language={codeLang} />
      )}
      {message.toolCalls.map((tc, i) => (
        <ToolCallView key={i} message={tc} />
      ))}
    </Box>
  )
}

// src/tui/input-bar.tsx
import React, { useState, useCallback } from 'react'
import { Box, Text, useInput } from 'ink'
import TextInput from 'ink-text-input'

interface InputBarProps {
  onSubmit: (text: string) => void
  isStreaming: boolean
}

export function InputBar({ onSubmit, isStreaming }: InputBarProps) {
  const [value, setValue] = useState('')
  const [history, setHistory] = useState<string[]>([])
  const [historyIndex, setHistoryIndex] = useState(-1)

  const handleSubmit = useCallback((text: string) => {
    if (!text.trim() || isStreaming) return
    setHistory(prev => [...prev, text])
    setHistoryIndex(-1)
    setValue('')
    onSubmit(text)
  }, [isStreaming, onSubmit])

  useInput((input, key) => {
    if (key.upArrow && history.length > 0) {
      const newIndex = Math.min(historyIndex + 1, history.length - 1)
      setHistoryIndex(newIndex)
      setValue(history[history.length - 1 - newIndex])
    }
    if (key.downArrow && historyIndex > 0) {
      const newIndex = historyIndex - 1
      setHistoryIndex(newIndex)
      setValue(history[history.length - 1 - newIndex])
    }
    if (key.downArrow && historyIndex === 0) {
      setHistoryIndex(-1)
      setValue('')
    }
  })

  return (
    <Box borderStyle="single" borderColor="gray" paddingX={1}>
      <Text color="green">❯ </Text>
      {isStreaming ? (
        <Text color="gray">等待 AI 回复中...</Text>
      ) : (
        <TextInput value={value} onChange={setValue} onSubmit={handleSubmit} placeholder="输入消息或 /help" />
      )}
    </Box>
  )
}

export function useAgent(deps: UseAgentDeps) {
  const [completedMessages, setCompletedMessages] = useState<Message[]>([])
  const [activeMessage, setActiveMessage] = useState<ActiveMessage | null>(null)
  const [isStreaming, setIsStreaming] = useState(false)
  const historyRef = useRef<ChatMessage[]>([])
  const abortRef = useRef<AbortController | null>(null)

  const sendMessage = useCallback(async (text: string) => {
    if (isStreaming) return

    // 1. 添加用户消息
    const userMsg: Message = { id: genId(), role: 'user', content: text, timestamp: Date.now() }
    setCompletedMessages(prev => [...prev, userMsg])

    // 2. 开始流式
    setIsStreaming(true)
    const activeId = genId()
    setActiveMessage({ id: activeId, content: '', toolCalls: [] })

    // 3. 创建 AbortController
    abortRef.current = new AbortController()

    // 4. 创建 AgentLoop 并运行
    const loop = new AgentLoop(deps.llm, deps.toolExecutor, {
      maxIterations: 50,
      maxTimeoutMs: 300000,
      systemPrompt: deps.systemPrompt,
      tools: deps.toolRegistry.getAll(),
      permManager: deps.permManager,
      hookManager: deps.hookManager,
      compactor: deps.compactor,
      onStreamChunk: (chunk: string) => {
        setActiveMessage(prev => prev ? { ...prev, content: prev.content + chunk } : null)
      },
      onToolCall: (toolName: string, args: Record<string, unknown>) => {
        setActiveMessage(prev => prev ? {
          ...prev,
          toolCalls: [...prev.toolCalls, { id: genId(), role: 'tool_call', content: '', toolName, toolArgs: args, timestamp: Date.now() }],
        } : null)
      },
    })

    try {
      const result = await loop.run(text, historyRef.current)

      // 5. 将活跃消息转为已完成
      setActiveMessage(prev => {
        if (prev) {
          const assistantMsg: Message = { id: prev.id, role: 'assistant', content: prev.content, timestamp: Date.now() }
          setCompletedMessages(completed => [...completed, assistantMsg, ...prev.toolCalls])
        }
        return null
      })
      setIsStreaming(false)

      // 6. 更新历史
      historyRef.current.push(
        { role: 'user', content: text },
        { role: 'assistant', content: result.finalText },
      )
    } catch (e: unknown) {
      // 保存已完成部分
      setActiveMessage(prev => {
        if (prev) {
          const assistantMsg: Message = { id: prev.id, role: 'assistant', content: prev.content || '(中断)', timestamp: Date.now() }
          setCompletedMessages(completed => [...completed, assistantMsg, ...prev.toolCalls])
        }
        return null
      })
      setIsStreaming(false)

      const errorMsg: Message = { id: genId(), role: 'assistant', content: `错误: ${(e as Error).message}`, timestamp: Date.now() }
      setCompletedMessages(prev => [...prev, errorMsg])
    }
  }, [isStreaming, deps])

  const interrupt = useCallback(() => {
    if (abortRef.current) {
      abortRef.current.abort()
    }
  }, [])

  const clearMessages = useCallback(() => {
    setCompletedMessages([])
    setActiveMessage(null)
    historyRef.current = []
  }, [])

  return { completedMessages, activeMessage, isStreaming, sendMessage, interrupt, clearMessages }
}
```

// src/tui/app.tsx
import React, { useState, useCallback } from 'react'
import { render, Box, Text, Static, useApp, useInput } from 'ink'
import { StatusBar } from './status-bar.js'
import { InputBar } from './input-bar.js'
import { MessageItem } from './message-item.js'
import { StreamMessage } from './stream-message.js'
import { useAgent, UseAgentDeps } from './use-agent.js'
import { Message } from './types.js'

interface AppProps extends UseAgentDeps {
  initialMode?: string
}

function AppInner({ initialMode = 'build', ...agentDeps }: AppProps) {
  const { exit } = useApp()
  const { completedMessages, activeMessage, isStreaming, sendMessage, interrupt, clearMessages } = useAgent(agentDeps)
  const [mode, setMode] = useState(initialMode)
  const [cost, setCost] = useState(0)

  const handleSubmit = useCallback((text: string) => {
    // 斜杠命令处理
    if (text.startsWith('/')) {
      handleSlashCommand(text)
      return
    }
    sendMessage(text)
  }, [sendMessage])

  const handleSlashCommand = useCallback((cmd: string) => {
    const parts = cmd.trim().split(/\s+/)
    const name = parts[0].slice(1)

    switch (name) {
      case 'help':
        const helpMsg: Message = {
          id: 'help',
          role: 'assistant',
          content: '可用命令: /help /new /mode /model /reload /cost /exit',
          timestamp: Date.now(),
        }
        // 直接添加到 completedMessages（通过 setCompletedMessages）
        break
      case 'new':
        clearMessages()
        break
      case 'mode':
        if (parts[1]) setMode(parts[1])
        break
      case 'exit':
      case 'quit':
        exit()
        break
      default:
        break
    }
  }, [clearMessages, exit])

  // 全局快捷键
  useInput((input, key) => {
    if (key.escape && isStreaming) {
      interrupt()
    }
  })

  return (
    <Box flexDirection="column" height="100%">
      <StatusBar
        model={agentDeps.llm.getModel() || 'unknown'}
        provider={agentDeps.llm.getProvider()}
        mode={mode}
        cost={cost}
        isStreaming={isStreaming}
      />
      <Box flexDirection="column" flexGrow={1} overflowY="hidden">
        <Static items={completedMessages}>
          {(msg: Message) => <MessageItem key={msg.id} message={msg} />}
        </Static>
        {activeMessage && <StreamMessage message={activeMessage} />}
      </Box>
      <InputBar onSubmit={handleSubmit} isStreaming={isStreaming} />
    </Box>
  )
}

/** 检测终端是否支持 Ink */
export function supportsInk(): boolean {
  return !!process.stdout.isTTY
    && !!process.stdout.columns
    && process.env.TERM !== 'dumb'
}

/** 渲染 TUI */
export function renderTUI(deps: UseAgentDeps & { initialMode?: string }) {
  const { waitUntilExit } = render(<AppInner {...deps} />, {
    exitOnCtrlC: false,
  })
  return waitUntilExit()
}
